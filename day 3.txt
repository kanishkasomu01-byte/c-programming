#include<stdio.h>
void swap(int *x,int *y){
    int temp;
    temp = *x;
    *x = *y;
    *y = temp;
    }

int main(){
        int x=65,y=54;
        printf("Before swapping\n");
         printf("The value of x:%d\n",x);
         printf("The value of y:%d\n",y);
 
         swap(&x,&y);
 
         printf("After swapping\n");
         printf("The value of x:%d\n",x);
         printf("The value of y:%d\n",y);
 
         return 0;
}




Before swapping
The value of x:65
The value of y:54
After swapping
The value of x:54
The value of y:65
